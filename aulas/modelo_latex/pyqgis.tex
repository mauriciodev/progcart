\begin{frame}{Arquitetura do QGIS}
    \textbf{QGIS é estruturado em camadas (layers de software):}
    
    \begin{itemize}
        \item \textbf{Core} → Modelo de dados geoespacial
        \item \textbf{GUI} → Interface gráfica (Qt)
        \item \textbf{Analysis} → Algoritmos e Processing
        \item \textbf{Providers} → OGR, GDAL, PostGIS, etc.
    \end{itemize}
    
    \vspace{0.4cm}
    
    \textbf{PyQGIS} é o binding Python da API C++ do QGIS.
\end{frame}

\begin{frame}[fragile]{QgsApplication e Inicialização}
    \textbf{Dentro do QGIS:}
    \begin{minted}{python}
# Já inicializado automaticamente
layer = iface.activeLayer()

    \vspace{0.4cm}

QgsApplication.setPrefixPath("/usr", True)
app = QgsApplication([], False)
app.initQgis()

# código aqui

app.exitQgis()
    \end{minted}
\end{frame}

\begin{frame}[fragile]{Manipulação da estrutura do Projeto no QGIS}
    \textbf{QgsProject}

    \vspace{0.3cm}

\begin{frame}[fragile]{PyQGIS — QgsProject.instance()}
    
    \textbf{QgsProject.instance()} fornece acesso ao \textbf{projeto QGIS atualmente aberto}.  
    É o ponto central para acessar \textbf{camadas, CRS, caminhos e estado do projeto}.
    
    
    \vspace{0.2cm}
    
    \begin{columns}
        
        \column{0.5\textwidth}
        \textbf{Gerenciamento de camadas}
        
        \begin{minted}{python}
project.mapLayers()

project.mapLayersByName("rios")

project.addMapLayer(layer)

project.removeMapLayer(layer.id())
        \end{minted}
        
        \column{0.5\textwidth}
        \textbf{Informações do projeto}
        
        \begin{minted}{python}
project.fileName()

project.homePath()

project.crs()

project.setCrs(crs)
        \end{minted}
        
    \end{columns}
    
    \vspace{0.3cm}
    
    \textbf{Exemplo: listar camadas}
    
    \begin{minted}{python}
for layer in project.mapLayers().values():
    print(layer.name())
    \end{minted}
    
\end{frame}

\begin{frame}[fragile]{Iterando Feições}

\begin{frame}[fragile]{Geometria (QgsGeometry)}
    
    \textbf{Operações geométricas}
    
    \begin{minted}{python}
from qgis.core import QgsGeometry

g1 = QgsGeometry.fromWkt("POINT(0 0)")
g2 = QgsGeometry.fromWkt("POINT(1 1)")


g_line = QgsGeometry.fromWkt("LINESTRING(0 0, 1 1, 2 1)")
print(g_line.length()) # comprimento 

g_poly = QgsGeometry.fromWkt(
"POLYGON((0 0,2 0,2 2,0 2,0 0))"
)
print(g_poly.area()) # area

# buffer
buffer_geom = g1.buffer(10, 8)

# operações espaciais
inter = g_poly.intersection(buffer_geom)
diff = g_poly.difference(buffer_geom)
    \end{minted}


    
\end{frame}

\begin{frame}[fragile]{Geometria (QgsGeometry): Criação}
    
    A classe \texttt{QgsGeometry} pode ser criada a partir de objetos
    \texttt{QgsPointXY}, listas de pontos ou anéis de polígonos.
    
    \vspace{0.3cm}
    
    \textbf{Exemplos}
    
    \begin{minted}{python}
from qgis.core import QgsGeometry, QgsPointXY

# POINT
p = QgsPointXY(-43.2, -22.9)
geom_point = QgsGeometry.fromPointXY(p)

# LINESTRING
line = [ QgsPointXY(0,0), QgsPointXY(1,1), QgsPointXY(2,1) ]
geom_line = QgsGeometry.fromPolylineXY(linha)

# POLYGON
ring = [[ QgsPointXY(0,0), QgsPointXY(2,0), QgsPointXY(2,2),
          QgsPointXY(0,2), QgsPointXY(0,0)
]]
geom_poly = QgsGeometry.fromPolygonXY(ring)
    \end{minted}
    
\end{frame}

\begin{frame}[fragile]{Geometria (QgsGeometry): Acesso às coordenadas}
    
    \textbf{Convertendo QgsGeometry para estruturas Python}
    
    \begin{minted}{python}
from qgis.core import QgsGeometry, QgsWkbTypes

# acessar coordenadas
p = g1.asPoint()
line = g_line.asPolyline()
poly = g_poly.asPolygon()

print(p)
print(line)
print(poly)
    \end{minted}
    
    \vspace{0.2cm}
    
    \textbf{Comparando o tipo da geometria}
    
    \begin{minted}{python}
geom = g_line
if geom.wkbType() == QgsWkbTypes.LineString:
    print("É uma linha")
if geom.wkbType() == QgsWkbTypes.Point:
    print("É um ponto")
if geom.wkbType() == QgsWkbTypes.Polygon:
    print("É um polígono")
    \end{minted}
    
\end{frame}

\begin{frame}[fragile]{Exemplo: Camada de memória com um ponto}
    
    Podemos criar camadas temporárias diretamente em memória usando \texttt{QgsVectorLayer}.
    
    \begin{minted}{python}
from qgis.core import (
QgsVectorLayer, QgsFeature, QgsGeometry, QgsPointXY, QgsProject
)

# criar camada de memória
layer = QgsVectorLayer("Point?crs=EPSG:4326", "Pontos", "memory")
# acessar provedor de dados
provider = layer.dataProvider()
# criar feição
feat = QgsFeature()
geom = QgsGeometry.fromPointXY(QgsPointXY(-43.2, -22.9))
feat.setGeometry(geom)
# adicionar feição na camada
provider.addFeature(feat)
# atualizar camada
layer.updateExtents()
# adicionar ao projeto
QgsProject.instance().addMapLayer(layer)
    \end{minted}
    
\end{frame}

\begin{frame}[fragile]{Exemplo: Camada de memória com linha geodésica}
    

    
    \begin{minted}{python}
from qgis.core import QgsDistanceArea
new_york = (40.661, -73.944) #lat lon
london = (51.5072, 0.1276) #lat lon
# Remember the order is X,Y
point1 = QgsPointXY(new_york[1], new_york[0])
point2 = QgsPointXY(london[1], london[0])
d = QgsDistanceArea()
d.setEllipsoid('WGS84')
# Create a geodesic line with vertices every 100km
vertices = d.geodesicLine(point1, point2, 100000)
# Create a polyline from the vertices
# The method returns 1 geometry unless the line crosses antimeridien
geodesic_line = QgsGeometry.fromPolylineXY(vertices[0])
# Create a line layer to display the route
vlayer = QgsVectorLayer('LineString?crs=EPSG:4326', 'route', 'memory')
provider = vlayer.dataProvider()
f = QgsFeature()
f.setGeometry(geodesic_line)
provider.addFeature(f)
vlayer.updateExtents() 
QgsProject.instance().addMapLayer(vlayer)
    \end{minted}
    
\end{frame}

\begin{frame}[fragile]{Transformação de CRS (PyQGIS)}
    
    O PyQGIS permite transformar geometrias entre sistemas de referência.
    
    \begin{minted}{python}
from qgis.core import (
QgsCoordinateReferenceSystem,
QgsCoordinateTransform,
QgsProject,
QgsGeometry
)
# geometria em WGS84
geom = QgsGeometry.fromWkt("POINT(-43 -22)")
# CRS origem e destino
src = QgsCoordinateReferenceSystem("EPSG:4326")
dst = QgsCoordinateReferenceSystem("EPSG:3857")
# criar transformador
transform = QgsCoordinateTransform(src, dst, QgsProject.instance())
# transformar geometria
geom.transform(transform)
print(geom.asWkt())
    \end{minted}
    
    Exemplo: conversão de coordenadas geográficas (lat/lon) para Web Mercator.
    
\end{frame}

\begin{frame}[fragile]{Estrutura Mínima de Script Standalone}
    \begin{minted}{python}
from qgis.core import *
from qgis.PyQt.QtWidgets import QApplication
import sys

QgsApplication.setPrefixPath(
    "/usr", True
)

app = QApplication(sys.argv)
qgs = QgsApplication([], False)
qgs.initQgis()

# --- código aqui ---

qgs.exitQgis()
    \end{minted}
\end{frame}
