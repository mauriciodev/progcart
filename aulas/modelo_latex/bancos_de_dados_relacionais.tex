\section{Conceitos básicos de banco de dados geográficos}
\begin{frame}{Conceitos Fundamentais}
    \textbf{Banco de Dados Relacional (BDR)}:
    
    \begin{itemize}
        \item Dados organizados em tabelas (relações)
        \item Linhas $\rightarrow$ tuplas
        \item Colunas $\rightarrow$ atributos
        \item Chaves primárias e estrangeiras
    \end{itemize}
    
    \vspace{0.4cm}
    
    Modelo baseado em:
    \begin{itemize}
        \item Álgebra relacional
        \item SQL (Structured Query Language)
    \end{itemize}
    
    \vspace{0.4cm}
    
    Exemplo de SGBD:
    \begin{itemize}
        \item \textbf{{PostgreSQL} {database system}} 
        \item \textbf{{MySQL} {database system}}
    \end{itemize}
\end{frame}


\begin{frame}[fragile]{Tabelas em bancos relacionais}
    
    Dados em bancos relacionais são organizados em \textbf{tabelas}.
    
    \begin{center}
        
        \begin{tikzpicture}

            \draw (0,2.5) -- (8,2.5);

    \begin{itemize}
        \item \textbf{Colunas} representam atributos
        \item \textbf{Linhas} representam registros
        \item Cada tabela normalmente possui uma \textbf{chave primária (ID)}
    \end{itemize}
    
\end{frame}

\begin{frame}{SQL - Structured Query Language}
    
    \textbf{SQL} é a linguagem padrão para manipulação de bancos de dados relacionais.
    
    \vspace{0.4cm}
    
    Principais usos:
    
    \begin{itemize}
        \item Consultar dados
        \item Inserir novos registros
        \item Atualizar registros existentes
        \item Remover dados
        \item Definir estrutura de tabelas
    \end{itemize}
    
    \vspace{0.4cm}
    
    Bancos que utilizam SQL:
    
    \begin{itemize}
        \item PostgreSQL 
        \item MySQL
        \item SQLite (e Geopackage)
        \item DuckDB
        \item SQL Server
    \end{itemize}
    
\end{frame}

docker exec -it postgis bash

    \begin{minted}{bash}
createdb geodb -U postgres
    \end{minted}

    \begin{minted}{bash}
psql -U postgres -d geodb
    \end{minted}

    \begin{minted}{bash}
docker exec -it postgis psql -U postgres -d geodb 
    \end{minted}     
\end{frame}

\begin{frame}[fragile]{Usuários e Permissões no PostgreSQL/PostGIS}

\textbf{1. Alterar senha do usuário \texttt{postgres}:}

\begin{minted}{sql}
ALTER USER postgres
WITH PASSWORD 'nova_senha';
\end{minted}

\vspace{0.3cm}

\textbf{2. Criar usuário \texttt{aluno}:}

\begin{minted}{sql}
CREATE USER aluno
WITH PASSWORD 'senha123';
\end{minted}

\vspace{0.3cm}

\textbf{3. Permitir conexão ao banco \texttt{geodb}:}

\begin{minted}{sql}
GRANT CONNECT ON DATABASE geodb TO aluno;
\end{minted}

\vspace{0.3cm}

\textbf{4. Dar permissões de leitura e escrita:}

\begin{minted}{sql}
GRANT USAGE ON SCHEMA public TO aluno;

GRANT SELECT, INSERT, UPDATE, DELETE
ON ALL TABLES IN SCHEMA public
TO aluno;
\end{minted}

\vspace{0.3cm}

\textbf{5. Aplicar permissões futuras automaticamente:}

\begin{minted}{sql}
ALTER DEFAULT PRIVILEGES IN SCHEMA public
GRANT SELECT, INSERT, UPDATE, DELETE
ON TABLES TO aluno;
\end{minted}

\end{frame}

\begin{frame}[fragile]{Estrutura básica de uma consulta}
    
    A operação mais comum em SQL é a consulta de dados usando \texttt{SELECT}.
    
    \begin{minted}{sql}
SELECT coluna1, coluna2
FROM tabela
WHERE condicao;
    \end{minted}
    
    Componentes:
    
    \begin{itemize}
        \item \textbf{SELECT} - define as colunas retornadas
        \item \textbf{FROM} - define a tabela consultada
        \item \textbf{WHERE} - filtra os registros
    \end{itemize}
    
    Exemplo:
    
    \begin{minted}{sql}
SELECT nome, populacao
FROM cidades
WHERE populacao > 3000000;
    \end{minted}
    
\end{frame}

\begin{frame}[fragile]{Inserindo, atualizando e removendo dados}
    
    SQL também permite modificar os dados.
    
    \vspace{0.3cm}

\vspace{0.3cm}

\begin{minted}{sql}
INSERT INTO cidades (nome, populacao, longitude, latitude)
VALUES ('Campinas', 1214000, -47.0608, -22.9056);

    \begin{minted}{sql}
UPDATE cidades

    \end{minted}

    \begin{minted}{sql}
DELETE FROM cidades

\begin{frame}[fragile]{Operações de agregação}
    
    Funções de agregação resumem dados.
    
    \begin{minted}{sql}
SELECT COUNT(*)
FROM estados;
    \end{minted}
    
    Funções comuns:
    
    \begin{itemize}
        \item \texttt{COUNT()} - número de registros
        \item \texttt{SUM()} - soma
        \item \texttt{AVG()} - média
        \item \texttt{MIN()} - valor mínimo
        \item \texttt{MAX()} - valor máximo
    \end{itemize}
    
    Exemplo:
    
    \begin{minted}{sql}
SELECT AVG(populacao)
FROM cidades;
    \end{minted}
    
\end{frame}

\begin{frame}[fragile]{Operações de agregação com GROUP BY e ORDER BY}

\textbf{Agrupando cidades por faixa populacional:}

\begin{minted}{sql}
SELECT "NM_REGIAO", count(*) AS quantidade_de_estados
FROM "BR_UF_2025"
GROUP BY "NM_REGIAO"
ORDER BY quantidade_de_estados;
\end{minted}

\vspace{0.3cm}

\textbf{Termos usados:}
\begin{itemize}
\item \textbf{AS}: cria um alias (apelido) para uma coluna ou tabela. Muito útil para referenciar colunas geradas por funções (Ex.: Count(*)).
\item \textbf{GROUP BY}: Agrupa valores iguais em uma coluna. Requer uma função de agregação depois (Ex.: Count() ).
\item \textbf{ORDER BY}: Ordena pelos valores de uma coluna.
\end{itemize}



\end{frame}

\begin{frame}[fragile]{Relacionando tabelas - JOIN}
    
    Bancos relacionais permitem combinar dados de várias tabelas.
    
    Exemplo de duas tabelas:
    
    \begin{itemize}
        \item cidades
        \item estados
    \end{itemize}
    
    Consulta com \textbf{JOIN}:
    
    \begin{minted}{sql}
SELECT c.nome, e.nome
FROM cidades c
JOIN estados e
ON c.estado_id = e.id;
    \end{minted}
    
    \textbf{JOIN} permite:
    
    \begin{itemize}
        \item combinar informações
        \item evitar duplicação de dados
        \item modelar relações entre entidades
    \end{itemize}
    
\end{frame}

\begin{frame}[fragile]{Ordem de execução de uma consulta SQL}
    
    
    \begin{columns}[t]
        \begin{column}[t]{0.48\textwidth}
            Embora a consulta seja escrita nesta ordem:
            \begin{minted}{sql}
SELECT nome, populacao
FROM cidades
WHERE populacao > 100000
ORDER BY populacao DESC;
            \end{minted}
            \begin{itemize}
                \item Primeiro o banco identifica a \textbf{tabela} (FROM)
                \item Depois aplica \textbf{filtros} (WHERE)
                \item Em seguida realiza \textbf{agrupamentos}
                \item Só então seleciona as \textbf{colunas finais}
                \item Por último ordena os resultados
            \end{itemize}
        \end{column}
        
        \begin{column}[t]{0.48\textwidth}
            O banco executa as operações aproximadamente nesta ordem:
            \begin{center}
                
                \begin{tikzpicture}[
                    node distance=1.0cm,
                    block/.style={rectangle, draw, rounded corners, minimum width=1cm, minimum height=0.8cm}
                    ]
                    
                    \node[block] (from) {FROM};
                    \node[block, below of=from] (where) {WHERE};
                    \node[block, below of=where] (group) {GROUP BY};
                    \node[block, below of=group] (having) {HAVING};
                    \node[block, below of=having] (select) {SELECT};
                    \node[block, below of=select] (order) {ORDER BY};
                    
                    \draw[->] (from) -- (where);
                    \draw[->] (where) -- (group);
                    \draw[->] (group) -- (having);
                    \draw[->] (having) -- (select);
                    \draw[->] (select) -- (order);
                    
                \end{tikzpicture}
                
            \end{center}
        \end{column}
    \end{columns}   
\end{frame}

\begin{frame}{Índices Espaciais: Por que são importantes?}
    
    Problemas comuns em consultas espaciais:
    
    \begin{itemize}
        \item Encontrar objetos que \textbf{intersectam} uma área
        \item Selecionar feições \textbf{próximas}
        \item Identificar objetos dentro de um \textbf{bounding box}
    \end{itemize}
    
    \textbf{Sem índice espacial:}
    
    \begin{itemize}
        \item Cada geometria precisa ser testada
        \item Complexidade aproximada: $O(n)$
    \end{itemize}
    
    \textbf{Com índice espacial:}
    
    \begin{itemize}
        \item Estrutura hierárquica de envelopes
        \item Redução drástica do número de testes geométricos
        \item Complexidade aproximada: $O(\log n)$
    \end{itemize}
    
    \vspace{0.4cm}
    
    \textbf{Ideia central:}
    \begin{itemize}
        \item Primeiro filtrar por \textbf{bounding boxes}
        \item Depois aplicar a operação geométrica precisa
    \end{itemize}
    
\end{frame}

\begin{frame}{Índices espaciais: Estrutura de um R-tree}
    
    \centering
    
    \begin{tikzpicture}[scale=0.9]
        
        % Espaço de dados
        \draw[thick] (0,0) rectangle (6,6);
        
        % Objetos
        \fill (1,1) circle (2pt);
        \fill (1.5,1.2) circle (2pt);
        \fill (4.5,1.5) circle (2pt);
        \fill (4.8,1.8) circle (2pt);
        \fill (2,4.5) circle (2pt);
        \fill (2.3,4.7) circle (2pt);
        
        % Bounding boxes
        \draw[red, thick] (0.7,0.8) rectangle (2.2,1.6);
        \draw[blue, thick] (4.2,1.2) rectangle (5.3,2.2);
        \draw[green!70!black, thick] (1.7,4.2) rectangle (2.8,5.2);
        
        % Nó superior
        \draw[dashed, thick] (0.4,0.6) rectangle (5.5,5.5);
        
        \node at (3,-0.7) {Objetos agrupados em \textbf{Minimum Bounding Rectangles (MBR)}};
        
    \end{tikzpicture}
    
    \vspace{0.3cm}
    
    \begin{itemize}
        \item Cada nó armazena o \textbf{retângulo mínimo} que cobre seus filhos
        \item Consultas percorrem apenas ramos relevantes
        \item Estrutura semelhante a uma árvore B para dados espaciais
    \end{itemize}
    
\end{frame}

    \end{itemize}

\begin{frame}{O que é DuckDB?}
    
    \textbf{DuckDB} é um banco de dados analítico embutido (embedded OLAP).
    
    OLAP (Online Analytical Processing)
    
    \vspace{0.4cm}
    
    Características principais:
    
    \begin{itemize}
        \item Arquivo único (.duckdb)
        \item Sem servidor (in-process)
        \item Execução vetorizada (vectorized engine)
        \item Otimizado para leitura intensiva (OLAP)
    \end{itemize}
    
    \vspace{0.4cm}
    
    Casos de uso típicos:
    
    \begin{itemize}
        \item Análise local de grandes datasets
        \item Ciência de dados
        \item Processamento de Parquet
        \item Integração com Python / R
    \end{itemize}
    
\end{frame}

\begin{frame}{DuckDB: Conceitos Arquiteturais Fundamentais}
    
    \textbf{1. Embedded Database}
    \begin{itemize}
        \item Executa dentro do processo Python
        \item Não há daemon ou serviço separado
    \end{itemize}
    
    \vspace{0.3cm}
    
    \textbf{2. Columnar Storage}
    \begin{itemize}
        \item Armazenamento orientado a colunas
        \item Ideal para agregações e scans analíticos
    \end{itemize}
    
    \vspace{0.3cm}
    
    \textbf{3. Integração com formatos modernos}
    \begin{itemize}
        \item Parquet (leitura direta)
        \item CSV
        \item Arrow
        \item Extensão Spatial (tipos GEOMETRY)
    \end{itemize}
    
    \vspace{0.3cm}
    
    \textbf{4. OLAP vs OLTP}
    \begin{itemize}
        \item Excelente para consultas analíticas
        \item Não projetado para alta concorrência transacional
    \end{itemize}
    
\end{frame}
