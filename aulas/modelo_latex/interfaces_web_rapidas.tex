\subsection{HTML + Flask}

\begin{frame}{HTML Form → Flask (WSGI)}
    
    \textbf{Fluxo da aplicação:}
    
    \begin{enumerate}
        \item Usuário preenche formulário HTML
        \item Navegador envia requisição HTTP (POST)
        \item Flask processa via rota (@app.route)
        \item Servidor retorna HTML como resposta
    \end{enumerate}
    
    \vspace{0.4cm}
    
    \textbf{Flask:}
    \begin{itemize}
        \item Microframework web em Python
        \item Baseado em WSGI
    \end{itemize}
    
    \vspace{0.4cm}
    
    Execução:
    \begin{itemize}
        \item \texttt{pixi add flask}
        \item \texttt{pixi run python app.py}
    \end{itemize}
    
\end{frame}

\begin{frame}[fragile]{Tipos de Requisição HTTP}

\textbf{Principais métodos HTTP:}

\begin{itemize}
    \item \textbf{GET} → Buscar dados (sem alterar estado)
    \item \textbf{POST} → Enviar dados / criar recurso
    \item \textbf{PUT} → Atualizar recurso (completo)
    \item \textbf{PATCH} → Atualizar parcialmente
    \item \textbf{DELETE} → Remover recurso
\end{itemize}

\textbf{Exemplos em Python:}

\begin{minted}{python}
import requests

# GET
r = requests.get("http://api.exemplo.com/users")

# POST
r = requests.post( "http://api.exemplo.com/users", json={"name": "Ana"} )

# PUT
r = requests.put( "http://api.exemplo.com/users/1", json={"name": "Ana Silva"})

# DELETE
r = requests.delete( "http://api.exemplo.com/users/1" )
\end{minted}

\vspace{0.3cm}

\textbf{Observações:}
\begin{itemize}
    \item GET é idempotente e seguro
    \item POST não é idempotente
    \item PUT e DELETE são idempotentes
\end{itemize}

\end{frame}

\begin{frame}[fragile]{Streamlit: Usuários com Duas Colunas}

\begin{minted}{python}
import streamlit as st

if "users" not in st.session_state: # Estado persistente
    st.session_state.users = [ {"id": 1, "name": "Ana"} ]

col1, col2 = st.columns(2) # Layout em duas colunas

with col1: # Coluna 1: Entrada
    st.subheader("Adicionar usuário")
    name = st.text_input("Nome")

    if st.button("Adicionar"):
        if name:
            new_user = {
                "id": len(st.session_state.users) + 1,
                "name": name
            }
            st.session_state.users.append(new_user)
            st.success(f"{name} adicionado!")

with col2: # Coluna 2: Visualização
    st.subheader("Lista de usuários")
    st.write(st.session_state.users)
\end{minted}
