\begin{frame}[fragile]{Trabalhando com Datas}

\textbf{Objetivo}

Armazenar e consultar datas em uma tabela PostgreSQL.

\vspace{0.3cm}

\textbf{Tabela de eventos}

\begin{minted}{sql}
CREATE TABLE eventos (
    id SERIAL PRIMARY KEY,
    descricao TEXT,
    data_evento DATE
);

INSERT INTO eventos
(descricao, data_evento)
VALUES
('Prova 1', '2025-09-15'),
('Entrega Trabalho', '2025-10-01'),
('Prova Final', '2025-12-10');
\end{minted}

\vspace{0.2cm}

\textbf{Consultar eventos após 01/10/2025}

\begin{minted}{sql}
SELECT *
FROM eventos
WHERE data_evento >= '2025-10-01';
\end{minted}

\vspace{0.2cm}

\textbf{Observação}

O PostgreSQL permite comparar datas utilizando operadores como
\texttt{<}, \texttt{<=}, \texttt{>} e \texttt{>=}.

\end{frame}

\begin{frame}[fragile]{Intervalos de Datas com DATERANGE}

\textbf{Objetivo}

Representar períodos utilizando um único campo.

\vspace{0.3cm}

\textbf{Tabela de férias}

\begin{minted}{sql}
CREATE TABLE ferias (
    id SERIAL PRIMARY KEY,
    funcionario TEXT,
    periodo DATERANGE
);

INSERT INTO ferias
(funcionario, periodo)
VALUES
('João',
 '[2025-01-10,2025-01-25]'),

('Maria',
 '[2025-07-01,2025-07-20]');
\end{minted}

\vspace{0.2cm}

\textbf{Quem estava de férias em 15/01/2025?}

\begin{minted}{sql}
SELECT funcionario
FROM ferias
WHERE periodo @> '2025-01-15';
\end{minted}

\vspace{0.2cm}

\textbf{Operador importante}

\begin{minted}{sql}
@>   contém
<@   está contido em
&&   possui interseção
\end{minted}

\end{frame}

\begin{frame}[fragile]{Chave Primária (Primary Key)}

\textbf{Conceito}

Uma chave primária identifica unicamente cada registro de uma tabela.

\vspace{0.3cm}

\textbf{Características}

\begin{itemize}
    \item Não pode ter valores repetidos
    \item Não pode ser nula
    \item Cada linha possui uma identificação única
\end{itemize}

\vspace{0.3cm}

\textbf{Exemplo}

\begin{minted}{sql}
CREATE TABLE estados (
    id SERIAL PRIMARY KEY,
    nome TEXT,
    sigla TEXT
);
\end{minted}

\vspace{0.2cm}

\textbf{Inserção de dados}

\begin{minted}{sql}
INSERT INTO estados (nome, sigla)
VALUES ('Paraná', 'PR'), ('São Paulo', 'SP');
\end{minted}

\vspace{0.2cm}

Neste exemplo, a coluna \texttt{id} identifica cada estado.

\end{frame}

% SLIDE 2
% =====================================================================
\begin{frame}[fragile]{Chave Estrangeira (Foreign Key)}

\textbf{Conceito}

Uma chave estrangeira cria um relacionamento entre duas tabelas.

\vspace{0.3cm}

\textbf{Exemplo}

\begin{minted}{sql}
CREATE TABLE cidades (
    id SERIAL PRIMARY KEY,
    nome TEXT,
    estado_id INTEGER,

    FOREIGN KEY (estado_id)
    REFERENCES estados(id)
);
\end{minted}

\vspace{0.2cm}

\textbf{Inserção de dados}

\begin{minted}{sql}
INSERT INTO cidades
(nome, estado_id)
VALUES
('Curitiba', 1),
('Campinas', 2);
\end{minted}

\vspace{0.3cm}

\textbf{Consulta relacionando as tabelas}

\begin{minted}{sql}
SELECT
    c.nome,
    e.sigla
FROM cidades c
JOIN estados e
ON c.estado_id = e.id;
\end{minted}

\vspace{0.2cm}

A coluna \texttt{estado\_id} referencia a chave primária da tabela \texttt{estados}.

\end{frame}

\begin{frame}{Relacionamentos N:M (Muitos para Muitos)}

\textbf{Problema}

Um aluno pode visitar vários estados.

\vspace{0.2cm}

Um estado pode ser visitado por vários alunos.

\vspace{0.4cm}

\textbf{Exemplos}

\begin{itemize}
    \item João visitou PR, SP e RJ.
    \item Maria visitou SP e MG.
    \item O estado de São Paulo foi visitado por João e Maria.
\end{itemize}

\vspace{0.4cm}

\textbf{Conclusão}

Nenhuma das tabelas pode armazenar diretamente essa relação.

É necessário criar uma \textbf{tabela de associação}.

\vspace{0.3cm}

\begin{center}
Aluno
$\leftrightarrow$
Ja\_Visitei
$\leftrightarrow$
Estado
\end{center}

\end{frame}


% =====================================================================
% SLIDE 2
% =====================================================================
\begin{frame}[fragile]{Implementação da Tabela de Associação}

\textbf{Tabela de alunos}

\begin{minted}{sql}
CREATE TABLE alunos (
    id SERIAL PRIMARY KEY, nome TEXT
);
\end{minted}

\textbf{Tabela de estados (já existente)}

\textbf{Tabela N:M}

\begin{minted}{sql}
CREATE TABLE ja_visitei (
    aluno_id INTEGER,
    estado_id INTEGER,

    PRIMARY KEY (
        aluno_id,
        estado_id
    ),
    FOREIGN KEY (aluno_id)
        REFERENCES alunos(id),
    FOREIGN KEY (estado_id)
        REFERENCES estados(id)
);
\end{minted}

\end{frame}

% SLIDE 3
% =====================================================================
\begin{frame}[fragile]{Inserindo e Consultando Relacionamentos}

\begin{columns}
\column[t]{0.49\textwidth}
\textbf{Dados de exemplo}

\begin{minted}{sql}
INSERT INTO alunos (nome)
VALUES
('João'),
('Maria');

INSERT INTO ja_visitei
(aluno_id, estado_id)
VALUES
(1, 1), -- João -> PR
(1, 2), -- João -> SP
(1, 3), -- João -> RJ
(2, 2), -- Maria -> SP
(2, 4); -- Maria -> MG
\end{minted}

\vspace{0.2cm}

\column[t]{0.49\textwidth}
\textbf{Quais estados João visitou?}

\begin{minted}{sql}
SELECT
    a.nome,
    e.sigla
FROM alunos a
JOIN ja_visitei j
    ON a.id = j.aluno_id
JOIN estados e
    ON e.id = j.estado_id
WHERE a.nome = 'João';
\end{minted}
\end{columns}
\vspace{0.2cm}

Resultado: PR, SP e RJ.

\end{frame}

\begin{frame}[fragile]{Exemplo: Criando a Coluna Geométrica}

\textbf{Enunciado}

A tabela possui colunas de longitude e latitude. Crie uma coluna espacial \texttt{geom} e preencha-a com geometrias do tipo ponto no sistema WGS84 (EPSG:4326).

\vspace{0.3cm}

\textbf{1. Adicionar a coluna geom}

\begin{minted}{sql}
ALTER TABLE cidades
ADD COLUMN geom geometry(Point, 4326);
\end{minted}

\vspace{0.2cm}

\textbf{2. Preencher a geometria a partir das coordenadas}

\begin{minted}{sql}
UPDATE cidades
SET geom = ST_SetSRID(
    ST_MakePoint(
        longitude,
        latitude
    ),
    4326
);
\end{minted}

\end{frame}

\begin{frame}[fragile]{Importando Shapefiles para PostGIS com GeoPandas}

\textbf{Objetivo}

Importar automaticamente todos os shapefiles contidos em um arquivo ZIP para o banco PostgreSQL/PostGIS.

\vspace{0.3cm}

\begin{minted}{python}
from pathlib import Path
import tempfile
import zipfile

import geopandas as gpd
from sqlalchemy import create_engine

# conexão com o banco
engine = create_engine(
    "postgresql+psycopg2://"
    "postgres:postgres@localhost/geodb"
)

arquivo_zip = "dados.zip"

with tempfile.TemporaryDirectory() as tmpdir:

    with zipfile.ZipFile(arquivo_zip) as z:
        z.extractall(tmpdir)

    for shp in Path(tmpdir).rglob("*.shp"):

        print(f"Importando {shp.name}")

        gdf = gpd.read_file(shp)

        tabela = shp.stem.lower()

        gdf.to_postgis(
            tabela,
            engine,
            if_exists="append",
            index=True
        )

print("Importação concluída.")
\end{minted}

\vspace{0.2cm}

\textbf{Resultado}

Cada shapefile será criado como uma tabela PostGIS utilizando o nome do arquivo.

\end{frame}
