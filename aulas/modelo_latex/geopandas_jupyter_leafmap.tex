\section{Cadernos jupyter}


\begin{frame}{Jupyter Notebooks — Conceito}
    
    Jupyter Notebook é um ambiente interativo para:
    
    \begin{itemize}
        \item Executar código (Python, R, etc.)
        \item Visualizar resultados imediatamente
        \item Combinar código, texto e gráficos
    \end{itemize}
    
    \vspace{0.4cm}
    
    \textbf{Muito usado em:}
    
    \begin{itemize}
        \item Ciência de dados
        \item Geoprocessamento
        \item Ensino e prototipagem
    \end{itemize}
    
    \vspace{0.4cm}
    
    Existem várias formas de usar:
    
    \begin{itemize}
        \item Google Colab
        \item VSCode / VSCodium
        \item JupyterLab local
    \end{itemize}
    
\end{frame}

\begin{frame}[fragile]{Comandos básicos no Jupyter Notebook}

    No Jupyter podemos combinar \textbf{código}, \textbf{comandos especiais} e \textbf{texto formatado}.
    
    \vspace{0.3cm}
    
    \begin{columns}[t]
    
    \column{0.48\textwidth}
    \textbf{Comandos de sistema (\texttt{!})}
    
    Executam comandos do terminal:
    
    \begin{minted}{python}
!ls
!pip install geopandas
!pwd
    \end{minted}
    
    \vspace{0.3cm}
    
    \textbf{Magic commands (\texttt{\%})}
    
    Comandos especiais do Jupyter:
    
    \begin{minted}{python}
%timeit sum(range(1000))
%matplotlib inline
    \end{minted}
    
    \column{0.48\textwidth}
    \textbf{Markdown}
    
    Permite escrever texto formatado.
    
    \begin{minted}{markdown}
# Título
## Seção

Texto com **negrito**
e *itálico*
    \end{minted}
    
    \vspace{0.3cm}
    
    \textbf{Figuras}
    
    \begin{minted}{markdown}
![Mapa](figura.png)
    \end{minted}
    
    \end{columns}
    
    \vspace{0.3cm}
    
    \textbf{Resumo}
    
    \begin{itemize}
    \item \texttt{!} → comandos do sistema
    \item \texttt{\%} → comandos do Jupyter
    \item Markdown → documentação e organização
    \end{itemize}

\end{frame}

\subsection{VSCodium Jupyter plugin}
\begin{frame}[fragile]{VSCode / VSCodium (plugin Jupyter)}
    
    Permite rodar notebooks dentro do editor.
    
    \vspace{0.3cm}
    
    \textbf{Instalação}
    \begin{columns}[t]
        \column{0.48\textwidth}
            \begin{minted}{bash}
# instalar python com pip
pip install notebook ipykernel
        \end{minted}
        \column{0.48\textwidth}
            \begin{minted}{bash}
# instalar python com pixi
pixi add notebook ipykernel
        \end{minted}
    \end{columns}

    
    Instalar extensões:
    
    \begin{itemize}
        \item Python
        \item Jupyter
    \end{itemize}
    
    \vspace{0.3cm}
    
    \textbf{Uso}
    
    \begin{itemize}
        \item Abrir ou criar um novo arquivo com extensão \texttt{.ipynb}.
        \item Executar células diretamente
    \end{itemize}
    
    \vspace{0.3cm}
    
    \textbf{Vantagens}
    
    \begin{itemize}
        \item Integração com código Python
        \item Debugging e autocomplete
        \item Ambiente local controlado
    \end{itemize}
    
\end{frame}

\begin{frame}[fragile]{Matplotlib: Exemplo de regressão linear}

Exemplo em um notebook Jupyter com dados sintéticos.

\vspace{0.3cm}

    \begin{minted}{python}
import numpy as np
import matplotlib.pyplot as plt

# gerar dados sintéticos
np.random.seed(0)
x = np.linspace(0, 10, 30)
y = 2.5 * x + 5 + np.random.randn(30)

# ajustar regressão linear (y = ax + b)
coef = np.polyfit(x, y, 1)
y_fit = np.polyval(coef, x)

# plot
plt.scatter(x, y, label="dados")
plt.plot(x, y_fit, label="ajuste linear")
plt.xlabel("x")
plt.ylabel("y")
plt.legend()
plt.title("Regressão linear")

plt.show()
    \end{minted}

\end{frame}

\begin{frame}[fragile]{Leafmap em caderno Jupyter: GeoDataFrame}
    
    \begin{minted}{python}
import leafmap
import geopandas as gpd

# criar mapa centrado no Brasil
m = leafmap.Map(center=[-15, -55], zoom=4)

# carregar dados vetoriais
url = "https://naciscdn.org/naturalearth/110m/physical/ne_110m_land.zip"
gdf = gpd.read_file(url)

# adicionar ao mapa com estilo
m.add_gdf(
    gdf,
    layer_name="Natural Earth",
    style={
        "color": "red",        # cor da borda
        "fillColor": "lightgray",
        "weight": 2
    }
)

# exibir mapa
m
    \end{minted}
    
\end{frame}

\begin{frame}[fragile]{Leafmap em caderno Jupyter: COG}

\textbf{COG (Cloud Optimized GeoTIFF)} permite leitura eficiente via HTTP.

\vspace{0.3cm}

\textbf{Código}

\begin{minted}{python}
import leafmap
m = leafmap.Map(center=[20, 0], zoom=2)

# URL de um COG público (exemplo Sentinel-2)
url = "https://github.com/opengeos/data/releases/download/raster/Libya-2023-07-01.tif"

# adicionar COG ao mapa
m.add_cog_layer(url, name="COG")

bounds = leafmap.cog_bounds(url)
m.zoom_to_bounds(bounds)
# exibir mapa
m
\end{minted}

\vspace{0.3cm}

\textbf{Vantagens do COG}

\begin{itemize}
\item Acesso direto via URL (sem download completo)
\item Leitura por blocos (otimizado para web)
\item Ideal para grandes rasters
\end{itemize}

\end{frame}
