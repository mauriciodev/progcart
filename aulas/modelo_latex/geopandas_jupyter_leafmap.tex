\section{Cadernos jupyter}


\begin{frame}{Jupyter Notebooks — Conceito}
    
    Jupyter Notebook é um ambiente interativo para:
    
    \begin{itemize}
        \item Executar código (Python, R, etc.)
        \item Visualizar resultados imediatamente
        \item Combinar código, texto e gráficos
    \end{itemize}
    
    \vspace{0.4cm}
    
    \textbf{Muito usado em:}
    
    \begin{itemize}
        \item Ciência de dados
        \item Geoprocessamento
        \item Ensino e prototipagem
    \end{itemize}
    
    \vspace{0.4cm}
    
    Existem várias formas de usar:
    
    \begin{itemize}
        \item Google Colab
        \item VSCode / VSCodium
        \item JupyterLab local
    \end{itemize}
    
\end{frame}

\begin{frame}[fragile]{Google Colab (na nuvem)}
    
    Executa notebooks diretamente no navegador.
    
    \vspace{0.3cm}
    
    \textbf{Como usar:}
    
    \begin{itemize}
        \item Acessar: \texttt{https://colab.research.google.com}
        \item Login com conta Google
        \item Criar novo notebook
    \end{itemize}
    
    \vspace{0.3cm}
    
    \textbf{Instalação de pacotes}
    
    \begin{minted}{python}
!pip install geopandas
    \end{minted}
    
    \textbf{Vantagens}
    
    \begin{itemize}
        \item Não precisa instalar nada
        \item Roda na nuvem (CPU/GPU)
        \item Fácil compartilhamento
    \end{itemize}
    
    \textbf{Limitações}
    
    \begin{itemize}
        \item Ambiente temporário
        \item Dependente de internet
    \end{itemize}
    
\end{frame}

\begin{frame}[fragile]{Instalação do Jupyter com Pixi}
    
    \textbf{Contexto:}  
    Pixi gerencia ambientes e dependências de projetos Python.
    
    \vspace{0.3cm}
    
    \textbf{1. Criar um projeto Pixi}
    \begin{minted}{bash}
pixi init projeto-jupyter
cd projeto-jupyter
    \end{minted}
    
    \vspace{0.2cm}
    
    \textbf{2. Adicionar Jupyter ao ambiente}
    \begin{minted}{bash}
pixi add jupyter
    \end{minted}
    
    \vspace{0.2cm}
    
    \textbf{3. Executar o Jupyter Notebook}
    \begin{minted}{bash}
pixi run jupyter notebook
    \end{minted}
    
    \vspace{0.3cm}
    
    \textbf{Observações:}
    \begin{itemize}
        \item Ambiente isolado e reprodutível
        \item Dependências descritas no \texttt{pixi.toml}
    \end{itemize}
    
\end{frame}
\begin{frame}[fragile]{Instalação do Jupyter com uv}
    
    \textbf{Contexto:}  
    \texttt{uv} é um gerenciador rápido de ambientes e pacotes Python.
    
    \vspace{0.3cm}
    
    \textbf{1. Criar ambiente virtual}
    \begin{minted}{bash}
uv venv
    \end{minted}
    
    \vspace{0.2cm}
    
    \textbf{2. Ativar o ambiente}
    \begin{minted}{bash}
source .venv/bin/activate
    \end{minted}
    
    \vspace{0.2cm}
    
    \textbf{3. Instalar Jupyter}
    \begin{minted}{bash}
uv pip install jupyter
    \end{minted}
    
    \vspace{0.2cm}
    
    \textbf{4. Executar o Jupyter Notebook}
    \begin{minted}{bash}
jupyter notebook
    \end{minted}
    
    \vspace{0.3cm}
    
    \textbf{Observações:}
    \begin{itemize}
        \item Fluxo similar ao \texttt{pip + venv}
        \item Foco em desempenho
    \end{itemize}
    
\end{frame}

\section{Visualização de geoinformação no Jupyter}

\begin{frame}[fragile]{Exemplo — Leafmap em caderno Jupyter}
    
    \textbf{Contexto:}  
    \texttt{leafmap} permite criar mapas interativos usando Python,
    integrando Jupyter + Leaflet.
    
    \vspace{0.3cm}
    
    \begin{minted}{python}
import leafmap

# Criar um mapa interativo
m = leafmap.Map(center=[-15, -55], zoom=4)

# Adicionar camada base
m.add_basemap("OpenStreetMap")

# Exibir o mapa no Jupyter
m
    \end{minted}
    
    \vspace{0.3cm}
    
    \textbf{O que acontece no notebook:}
    \begin{itemize}
        \item O mapa é renderizado interativamente
        \item Zoom e navegação com o mouse
        \item Ideal para exploração espacial
    \end{itemize}
    
\end{frame}

\begin{frame}[fragile]{Exemplo: Mapa interativo com Leafmap}
    
    \textbf{Leafmap} permite criar mapas interativos facilmente em notebooks.
    
    \vspace{0.3cm}
    
    \textbf{Instalação (Colab)}
    
    \begin{minted}{python}
!pip install leafmap geopandas
    \end{minted}
    
    \textbf{Código}
    
    \begin{minted}{python}
import leafmap
import geopandas as gpd

# criar mapa centrado no Brasil
m = leafmap.Map(center=[-15, -55], zoom=4)

# carregar dados vetoriais
url = "https://raw.githubusercontent.com/" \
"geopandas/geopandas/main/examples/" \
"naturalearth_lowres/naturalearth_lowres.shp"

gdf = gpd.read_file(url)

# filtrar Brasil
br = gdf[gdf.name == "Brazil"]

# adicionar ao mapa
m.add_gdf(br, layer_name="Brasil")

# exibir mapa
m
    \end{minted}
    
\end{frame}
